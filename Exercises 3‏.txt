#include <bits/stdc++.h>
using namespace std;

struct Node
{
    int data;
    Node *next;
};

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    Node *head = new Node();
    Node *temp = head;

    for (int i = 1; i <= 4; i++)
    {
        temp->data = i;
        if (i != 4)
        {
            temp->next = new Node();
            temp = temp->next;
        }
        else
        {
            temp->next = NULL;
        }
    }

    Node *print = head;
    while (print != NULL)
    {
        cout << print->data << " ";
        print = print->next;
    }

    return 0;
}
