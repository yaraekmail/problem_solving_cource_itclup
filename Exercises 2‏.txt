void q1(int x)
{
    if (x >= 0)
    {
        if (x % 2 == 0)
            cout << "positive even";
        else
            cout << "positive odd";
    }
    else
        cout << "NEGATIVE";
}
void q2(int x)
{
    if (x <= 100)
    {
        int b = x * .1;
        int c = x - b;
        cout << c;
    }
    else
        cout << x;
}
void q3(int x)
{
    if (x % 400 == 0)
    {
        cout << x << " is leap year";
    }
    else
    {
        if (x % 4 == 0 && x % 100 != 0)
            cout << x << " is leap year";
        else
            cout << x << " is  not leap year";
    }
}
